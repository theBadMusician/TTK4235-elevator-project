9void hmi_stopBtnHandler(ElevatorState* elev) {
10 if (elev->currentState == STATE_INIT) {
11 elev->isStopPressed = false;
12 return;
13 }
14
15 elev->isStopPressed = elevio_stopButton();
16
17 if (!elev->isStopPressed && timer_isTimeout(&elev->stopDebouncerTimer)) {
18 elevio_stopLamp(false);
19
20 // Force reset the timer using stop
21 timer_stop(&elev->stopDebouncerTimer);
22
23 // Debounce is finished
24 elev->isStopDebouncing = false;
25 }
26 else if (elev->isStopPressed) {
27 elevio_stopLamp(true);
28
29 // Reset the timer using stop and debounce
30 elev->isStopDebouncing = true;
31 timer_stop(&elev->stopDebouncerTimer);
32 timer_start(&elev->stopDebouncerTimer, 0.5);
33
34 // Transition to stop state immediately
35 elev->currentState = STATE_STOP;
36 }
37}

50void hmi_orderHandler(ElevatorState* elev) {
51 bool isStopClear = !elev->isStopPressed && !elev->isStopDebouncing;
52 // Check for button presses
53 for(int f = 0; f < N_FLOORS; f++) {
54 for(int b = 0; b < N_BUTTONS; b++) {
55 bool isPressed = elevio_callButton(f, b);
56
57 // Rising Edge Detection
58 if (isPressed && !elev->prevBtnStates[f][b] && isStopClear) {
59 elevio_buttonLamp(f, b, true);
60 elev->orderArr[f][b] = true;
61 }
62 // Save current state for the next loop
