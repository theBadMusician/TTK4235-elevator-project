15/**
16 * @brief Represents the current operational state of the elevator.
17 */
18typedef enum {
19 STATE_INIT, /**< Initialization state upon startup. */
20 STATE_IDLE, /**< Elevator is stationary and awaiting orders. */
21 STATE_MOVING, /**< Elevator is currently moving between floors. */
22 STATE_DOOR_OPEN, /**< Elevator has reached a target floor and doors are open. */
23 STATE_STOP /**< Emergency stop state triggered by the stop button. */
24} State;

26/**
27 * @brief Central structure containing all state information for the elevator.
28 * This structure encapsulates the physical state (floor, direction),
29 * the internal logic state (current operation, active orders), and the required
30 * timers for asynchronous events like door operations and switch debouncing.
31 */
32typedef struct {
33 // States
34 State currentState; /**< The active operational state of the elevator. */
35 int currentFloor; /**< Most recently registered floor sensor. */
36 MotorDirection currentDir; /**< Current direction of motor movement. */
37
38 // Previous states
39 int prevFloor; /**< The floor the elevator was at previously. */
40 MotorDirection prevDir; /**< The previous direction of movement. */
41
42 // Queue order array
43 bool orderArr[N_FLOORS][N_BUTTONS]; /**< Matrix tracking all active floor and cabin orders. */
44
45 // Btn states
46 bool prevBtnStates[N_FLOORS][N_BUTTONS]; /**< Matrix tracking previous button states for edge detection. */
47
48 // Stop button
49 bool isStopped; /**< Flag indicating if the elevator is currently halted by the stop button. */
50 bool isStopPressed; /**< Flag indicating if the physical stop button is currently held down. */
51 bool isStopDebouncing; /**< Flag indicating if the stop button release is currently being debounced. */
52
53 // Timers
54 timerAlarm doorOpenTimer; /**< Timer regulating how long the doors remain open. */
55 timerAlarm btnQueryTimer; /**< Timer regulating the polling rate for button presses. */
56 timerAlarm stopDebouncerTimer; /**< Timer handling the debounce delay for the stop button. */
57} ElevatorState;

59/**
60 * @brief Initializes the elevator state structure with default values.
61 *
62 * @param[out] state Pointer to the ElevatorState structure to be initialized.
63 *
64 * @pre The `state` pointer must not be NULL.
65 * @post All internal variables are set to their safe startup defaults (e.g., STATE_INIT,
66 * order arrays cleared, timers initialized).
67 *
68 * @details Sets the initial state to STATE_INIT and defaults the floor tracking
69 * to 0. Motor directions are explicitly halted using DIRN_STOP. The order queue
70 * and previous button state arrays are iteratively cleared to ensure no "ghost"
71 * inputs exist in memory upon startup. Finally, the required hardware/software
72 * timers are initialized for later use.
73 */
74void elevatorState_init(ElevatorState* state);
