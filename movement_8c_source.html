25bool movement_recoveryHandler(ElevatorState* elev) {
26 /* Recovery when stopped between floors */
27 if (elev->currentFloor == -1 && elev->isStopped) {
28 // If there are no orders, wait
29 if (!movement_requestsAbove(elev, -1)) {
30 return true;
31 }
32
33 // Check if there is an order specifically at the last floor
34 bool orderAtPrevFloor = elev->orderArr[elev->prevFloor][BUTTON_CAB] ||
35 elev->orderArr[elev->prevFloor][BUTTON_HALL_UP] ||
36 elev->orderArr[elev->prevFloor][BUTTON_HALL_DOWN];
37
38 // Route directly towards the requested floor
39 if (elev->prevDir == DIRN_UP) {
40 // Hovering above prevFloor
41 if (movement_requestsAbove(elev, elev->prevFloor)) {
42 elev->currentDir = DIRN_UP;
43 } else if (orderAtPrevFloor || movement_requestsBelow(elev, elev->prevFloor)) {
44 elev->currentDir = DIRN_DOWN;
45 }
46 }
47 else if (elev->prevDir == DIRN_DOWN) {
48 // Hovering below prevFloor
49 if (movement_requestsBelow(elev, elev->prevFloor)) {
50 elev->currentDir = DIRN_DOWN;
51 } else if (orderAtPrevFloor || movement_requestsAbove(elev, elev->prevFloor)) {
52 elev->currentDir = DIRN_UP;
53 }
54 }
55
56 elevio_motorDirection(elev->currentDir);
57 elev->isStopped = false;
58 return true;
59 }
60
61 // Not in a recovery state
62 return false;
63}

65bool movement_stopHandler(ElevatorState* elev) {
66 /* Check if we should stop at the current floor */
67
68 // Always stop for cab calls at the current floor
69 if (elev->orderArr[elev->currentFloor][BUTTON_CAB]) return true;
70
71 if (elev->currentDir == DIRN_UP) {
72 // Stop for UP hall calls, or for DOWN calls if it is the highest request
73 if (elev->orderArr[elev->currentFloor][BUTTON_HALL_UP] || (!movement_requestsAbove(elev, elev->currentFloor) && elev->orderArr[elev->currentFloor][BUTTON_HALL_DOWN])) {
74 return true;
75 }
76 }
77 else if (elev->currentDir == DIRN_DOWN) {
78 // Stop for DOWN hall calls, or for UP calls if it is the lowest request
79 if (elev->orderArr[elev->currentFloor][BUTTON_HALL_DOWN] || (!movement_requestsBelow(elev, elev->currentFloor) && elev->orderArr[elev->currentFloor][BUTTON_HALL_UP])) {
80 return true;

86void movement_directionHandler(ElevatorState* elev) {
87 /* Determine next travel direction and apply it to the hardware */
88 if (elev->currentDir == DIRN_UP) {
89 if (movement_requestsAbove(elev, elev->currentFloor)) {
90 elev->currentDir = DIRN_UP;
91 } else if (movement_requestsBelow(elev, elev->currentFloor)) {
92 elev->currentDir = DIRN_DOWN;
93 } else {
94 elev->currentDir = DIRN_STOP;
95 elev->currentState = STATE_IDLE;
96 }
97 }
98 else if (elev->currentDir == DIRN_DOWN) {
99 if (movement_requestsBelow(elev, elev->currentFloor)) {
100 elev->currentDir = DIRN_DOWN;
101 } else if (movement_requestsAbove(elev, elev->currentFloor)) {
102 elev->currentDir = DIRN_UP;
103 } else {
104 elev->currentDir = DIRN_STOP;
105 elev->currentState = STATE_IDLE;
106 }
107 }
108 // Wake up from a stopped state
109 else if (elev->currentDir == DIRN_STOP) {
110 if (movement_requestsAbove(elev, elev->currentFloor)) {
111 elev->currentDir = DIRN_UP;
112 } else if (movement_requestsBelow(elev, elev->currentFloor)) {
113 elev->currentDir = DIRN_DOWN;
114 } else {
115 elev->currentState = STATE_IDLE;
116 }
117 }
118
119 // Apply the direction
120 if (elev->currentDir != DIRN_STOP) {
121 elev->prevDir = elev->currentDir;
122 }
123
124 // Engage the motor
125 elevio_motorDirection(elev->currentDir);
