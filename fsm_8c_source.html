17/* Initializes hardware and forces the elevator to move DOWN until a valid
18 floor is found to establish a known starting position. */
19static void fsm_onInit(void) {
20 // Driver and state init
21 log_info("Initializing elevator...");
22 elevio_init();
23 elevatorState_init(&elevator);
24
25 // Clear button lights
26 for(int f = 0; f < N_FLOORS; f++){
27 for(int b = 0; b < N_BUTTONS; b++){
28 elevio_buttonLamp(f, b, false);
29 }
30 }
31
32 // Localize a floor (PRD: O1)
33 // Blocking, ignores everything else (PRD: O2)
34 while (elevio_floorSensor() == -1) {
35 elevio_motorDirection(DIRN_DOWN);
36 timer_msSleep(1);
37 }
38 elevio_motorDirection(DIRN_STOP);
39
40 // Init floor state
41 elevator.prevFloor = elevio_floorSensor();
42 elevator.currentFloor = elevio_floorSensor();
43
44 // State transition T_0: STATE_INIT -> STATE_IDLE
45 elevator.currentState = STATE_IDLE;
46}

48/* Halts motor and closes doors. Transitions to MOVING if any order is detected. */
49static void fsm_onIdle(void) {
50 // Set the current direction (movement moment) to stop

62/* Handles routing. Delegates recovery and stop evaluation to the movement module.
63 Transitions to DOOR_OPEN if the elevator reaches a target floor. */
64static void fsm_onMoving(void) {
65 // Attempt recovery if stopped mid-shaft
66 if (movement_recoveryHandler(&elevator)) {
67 return;
68 }
69
70 // Normal operation requires a valid floor reading
71 if (elevator.currentFloor == -1) {
72 return;
73 }
74
75 // Evaluate if there is an order to stop at this floor
76 if (movement_stopHandler(&elevator)) {
77 elevio_motorDirection(DIRN_STOP);
78 elevator.currentState = STATE_DOOR_OPEN;
79 return;
80 }
81
82 // Determine and apply the next movement direction
83 movement_directionHandler(&elevator);
84}

86/* Manages the 3-second door open delay. The timer resets if an obstruction
87 or stop button is detected. Clears orders for the current floor upon completion. */
88static void fsm_onDoorOpen(void) {
89 if (timer_isTimeout(&elevator.doorOpenTimer) && !elevio_obstruction() && !elevio_stopButton()) {
90 timer_stop(&elevator.doorOpenTimer);
91 elevio_doorOpenLamp(false);
92#ifdef DEBUG
93 log_debug("Door closed!");
94#endif // DEBUG
95
96 // Clear orders for this floor
97 for (int b = 0; b < N_BUTTONS; b++) {
98 if (elevator.currentFloor == -1) {
99 elevator.orderArr[elevator.prevFloor][b] = false;
100 elevio_buttonLamp(elevator.prevFloor, b, false);
101 } else {
102 elevator.orderArr[elevator.currentFloor][b] = false;
103 elevio_buttonLamp(elevator.currentFloor, b, false);
104 }
105 }
106
107 // No more orders, transition to idle
108 if (areAllElementsFalse(elevator.orderArr)) elevator.currentState = STATE_IDLE;
109 // More orders await, move more
110 else elevator.currentState = STATE_MOVING;
111
112 // Just finished opening the door, return
113 return;
114 }
115 elevio_doorOpenLamp(true);
116#ifdef DEBUG
117 log_debug("Opening door for 3 seconds");
118#endif // DEBUG
119
120 // Restart timer if obstruction is present (PRD: D4)
121 // Restart timer if stop button is pressed (PRD: D3)
122 if (elevio_obstruction() || elevio_stopButton()) timer_stop(&elevator.doorOpenTimer);
123 timer_start(&elevator.doorOpenTimer, 3);
124}

126/* Immediate emergency halt. Wipes the order queue, clears lamps, and opens
127 * the door if the elevator is safely positioned at a floor. */
128static void fsm_onStop(void) {
129 /* Stop the elevator, clear queue, reset state */
130
131 // Set the current direction to stop and stop motor
132 elevator.currentDir = DIRN_STOP;
133 elevio_motorDirection(elevator.currentDir);
134
135 // Clear the order queue and btn states
136 memset(elevator.orderArr, false, sizeof(elevator.orderArr));
137 memset(elevator.prevBtnStates, false, sizeof(elevator.prevBtnStates));
138
139 // Clear button lights
140 for(int f = 0; f < N_FLOORS; f++){
141 for(int b = 0; b < N_BUTTONS; b++){
142 elevio_buttonLamp(f, b, false);
143 }
144 }
145
146 // state transition
147 if (elevator.currentFloor != -1) {
148 elevio_stopLamp(true);
149 elevio_doorOpenLamp(true); // Open door right away
150 elevator.currentState = STATE_DOOR_OPEN;
151 } else elevator.currentState = STATE_IDLE;
152
153 // Set stopped param for undefined floor escape
154 if (elevator.currentFloor == -1) elevator.isStopped = true;
155}

157void fsm_spin(void) {
158 // Update floor once per spin
159 hmi_floorHandler(&elevator);
160
161 // Check stop btn
162 hmi_stopBtnHandler(&elevator);
163
164 // Start querying after init
165 if (elevator.currentState != STATE_INIT) {
166
167 // Query slower than the spin loop
168 timer_start(&elevator.btnQueryTimer, 0.01);
169 if (timer_isTimeout(&elevator.btnQueryTimer)) {
170 // Reset the debounce timer
171 timer_stop(&elevator.btnQueryTimer);
172
173 // Handle btns and set orders
174 hmi_orderHandler(&elevator);
175 }
176 }
177
178 switch (elevator.currentState) {
179 case STATE_INIT: fsm_onInit(); break;
180 case STATE_IDLE: fsm_onIdle(); break;
181 case STATE_MOVING: fsm_onMoving(); break;
182 case STATE_DOOR_OPEN: fsm_onDoorOpen(); break;
183 case STATE_STOP: fsm_onStop(); break;
184 }
185
186 // Set the previous floor if not between floors
187 if (elevator.currentFloor != -1) elevator.prevFloor = elevator.currentFloor;
188}
